import java.util.Scanner;

// Base class for Player
class Player {
    String name;
    int age;
    String skill;
}

// Football player class extending Player
class Football extends Player {
}

// Hockey player class extending Player
class Hockey extends Player {
}

// Cricket player class extending Player
class Cricket extends Player {
}

public class PlayerDetails {
    public static void main(String[] args) {
        Scanner read = new Scanner(System.in);
        Football fb = new Football();
        Hockey hk = new Hockey();
        Cricket ckt = new Cricket();

        int op1, op2;

        while (true) {
            // Main menu
            System.out.println("\n1. Enter Details");
            System.out.println("2. Display Details");
            System.out.println("3. Exit");
            System.out.print("Enter your option: ");
            op1 = read.nextInt();

            switch (op1) {
                case 1: // Enter details
                    System.out.println("\n1. Football Player 2. Hockey Player 3. Cricket Player");
                    System.out.print("Enter your option: ");
                    op2 = read.nextInt();

                    switch (op2) {
                        case 1: // Football Player
                            System.out.print("Enter Name: ");
                            fb.name = read.next();
                            System.out.print("Enter Age: ");
                            fb.age = read.nextInt();
                            System.out.print("Enter Skill: ");
                            fb.skill = read.next();
                            break;

                        case 2: // Hockey Player
                            System.out.print("Enter Name: ");
                            hk.name = read.next();
                            System.out.print("Enter Age: ");
                            hk.age = read.nextInt();
                            System.out.print("Enter Skill: ");
                            hk.skill = read.next();
                            break;

                        case 3: // Cricket Player
                            System.out.print("Enter Name: ");
                            ckt.name = read.next();
                            System.out.print("Enter Age: ");
                            ckt.age = read.nextInt();
                            System.out.print("Enter Skill: ");
                            ckt.skill = read.next();
                            break;

                      
                    }
                    break;

                case 2: // Display details
                    System.out.println("\n1. Football Player 2. Hockey Player 3. Cricket Player");
                    System.out.print("Enter your option: ");
                    op2 = read.nextInt();

                    switch (op2) {
                        case 1: // Display Football Player
                            System.out.println("Name: " + fb.name);
                            System.out.println("Age: " + fb.age);
                            System.out.println("Skill: " + fb.skill);
                            break;

                        case 2: // Display Hockey Player
                            System.out.println("Name: " + hk.name);
                            System.out.println("Age: " + hk.age);
                            System.out.println("Skill: " + hk.skill);
                            break;

                        case 3: // Display Cricket Player
                            System.out.println("Name: " + ckt.name);
                            System.out.println("Age: " + ckt.age);
                            System.out.println("Skill: " + ckt.skill);
                            break;

                       
                    }
                    break;

                case 3: // Exit
                    System.out.println("Exiting...");
                    read.close();
                    return;

  
            }
}
}
}
