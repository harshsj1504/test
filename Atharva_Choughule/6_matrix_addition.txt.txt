import java.util.Scanner;

public class MatrixAddition {

    public static void main(String[] args) {
        // Declaring variables for row and column count
        int row, col;
        
        // Initializing three 2D arrays to store matrix data
        int a[][] = new int[20][20]; // Matrix a
        int b[][] = new int[20][20]; // Matrix b
        int c[][] = new int[20][20]; // Matrix c (result of addition of a and b)

        // Creating a Scanner object to read input
        Scanner sc = new Scanner(System.in);

        // Asking user to input the number of rows and columns
        System.out.println("Enter number of rows and columns");
        row = sc.nextInt();
        col = sc.nextInt();

        // Taking input for elements of matrix a
        System.out.println("Enter elements of matrix a:");
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                System.out.print("Enter element a[" + i + "][" + j + "]: ");
                a[i][j] = sc.nextInt();
            }
        }

        // Taking input for elements of matrix b
        System.out.println("Enter elements of matrix b:");
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                System.out.print("Enter element b[" + i + "][" + j + "]: ");
                b[i][j] = sc.nextInt();
            }
        }

        // Performing matrix addition and storing result in matrix c
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                c[i][j] = a[i][j] + b[i][j];
            }
        }

        // Displaying the resultant matrix c
        System.out.println("The resultant matrix after addition is:");
        for (int i = 0; i < row; i++) {
            for (int j = 0; j < col; j++) {
                System.out.print(c[i][j] + " ");
            }
            System.out.println(); // Newline after each row
        }

        // Closing the Scanner object to free resources
        sc.close();
    }
}
