import java.io.File;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;

class exp_11 {
    public static void main(String[] args) {
        File inf = new File("A.txt"); // Input file
        File outf = new File("B.txt"); // Output file
        FileReader ins = null;
        FileWriter outs = null;

        try {
            ins = new FileReader(inf);
            outs = new FileWriter(outf);
            int ch;

            // Reading characters from A.txt and writing them to B.txt
            while ((ch = ins.read()) != -1) {
                outs.write(ch);
            }
            
            System.out.println("File Copied");

        } catch (IOException e) {
            System.out.println("An error occurred: " + e);
            System.exit(-1);
        } finally {
            // Ensuring the resources are closed properly
            try {
                if (ins != null) ins.close();
                if (outs != null) outs.close();
            } catch (IOException e) {
                System.out.println("An error occurred while closing the files: " + e);
            }
 }
}
}
